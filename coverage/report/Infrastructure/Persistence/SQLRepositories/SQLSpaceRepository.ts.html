import { prisma } from "../PrismaClient"
import { RESERVABILITYCATEGORY, BUILDING, ROLE } from '@Domain/Constants';
import ISpaceRepository from '@/Domain/Repositories/ISpaceRepository';
import Space from "@/Domain/Entities/Space";
import ReservabilityCategory from "@/Domain/ValueObjects/ReservabilityCategory";
import Person from "@/Domain/Entities/Person";
import Role from "@/Domain/ValueObjects/Role";
import ReservationPolicy from "@/Domain/Policies/ReservationPolicy";
 
/**
 * SQL implementation of the space repository.
 */
class SQLSpaceRepository implements ISpaceRepository {
    /**
     * Finds spaces by various filters.
     * @param id - (Optional) The ID of the space.
     * @param reservabilityCategory - (Optional) The reservability category.
     * @param maxOccupants - (Optional) The maximum number of occupants.
     * @param floor - (Optional) The floor of the space.
     * @param reservable - (Optional) Whether the space is reservable.
     * @returns A promise resolving to an array of Space entities.
     */
    async find(id?: string, reservabilityCategory?: RESERVABILITYCATEGORY, maxOccupants?: number, floor?: string, reservable?: boolean): Promise<Space[]> {
        let prismaSpaces = await prisma.space.findManyWhere({
            where: {
                ...(id && { id }),
                ...(reservabilityCategory && { reservabilityCategory }),
                ...(maxOccupants && { maxOccupants }),
                ...(floor && { floor }),
                ...(reservable && { reservable }),
            },
        });
        // You should map to Space entities here if needed
        return prismaSpaces;
    }
 
    /**
     * Updates a space's properties.
     * @param id - The ID of the space to update.
     * @param space - The properties to update.
     * @returns A promise resolving to the updated Space entity.
     */
    async update(id: string, space: {
        reservable?: boolean, maxUsage?: number, openTime?: Date, closeTime?: Date,
        reservabilityCategory?: ReservabilityCategory, assignedTo?: Person | string | null
    }): Promise<Space> {
 
 
        const spaceUpdated = await prisma.$transaction(async (tx) => {
            let prismaSpace = await prisma.space.update({
                where: { id },
                data: {
                    ...(space.reservable !== undefined && { reservable: space.reservable }),
                    ...(space.maxUsage !== undefined && { maxUsage: space.maxUsage }),
                    ...(space.openTime !== undefined && { openTime: space.openTime }),
                    ...(space.closeTime !== undefined && { closeTime: space.closeTime }),
                    ...(space.reservabilityCategory && { reservabilityCategory: space.reservabilityCategory.getName() }),
                    ...((space.assignedTo || space.assignedTo === null) && { assignedTo: typeof space.assignedTo === "string" ? space.assignedTo : null }),
                    ...((space.assignedTo || space.assignedTo === null) && { assignedToPersonId: space.assignedTo instanceof Person ? space.assignedTo.toObject().id : null }),
                    ...((space.assignedTo || space.assignedTo === null) && { assignedToBuildingId: typeof space.assignedTo === "string" && space.assignedTo === BUILDING ? space.assignedTo : null }),
                }
            });
            const prismaBuilding = await tx.building.findMany({})
            const building = prismaBuilding[0];
            let allreservations = await tx.reservation.findMany({ where: { NOT: { state: "DONE" } }, include: { spaces: true, person: true } });
            for (const reservation of allreservations) {
                const prismaPerson = reservation.person
                let person = new Person(prismaPerson.id, prismaPerson.name, prismaPerson.email, new Role(prismaPerson.role as ROLE), prismaPerson.department);
                const spaces = await tx.space.findMany({
                    where: { id: { in: reservation.spaces.map(item => item.spaceId) } },
                });
 
                const spaceEntities = spaces.map(space => {
                    return new Space(
                        space.id,
                        space.name,
                        space.reservable,
                        space.maxOccupants,
                        space.maxUsage ? space.maxUsage : building.maxUsage,
                        space.openTime ? space.openTime : building.openTime,
                        space.closeTime ? space.closeTime : building.closeTime,
                        building.id,
                        space.floor,
                        space.size,
                        new ReservabilityCategory(space.reservabilityCategory as RESERVABILITYCATEGORY),
                        space.type,
                        space.assignedTo ?? space.assignedToBuildingId ?? (space.assignedToPersonId ? space.assignedToPersonId : "Unassigned"),
                    );
                });
 
                const can = ReservationPolicy.canReserve(person, spaceEntities, reservation.startTime, reservation.duration, reservation.maxAttendees, building.holidays)
                if (can.canReserve) {
                    await tx.reservation.update({
                        where: { id: reservation.id },
                        data: { state: "ACTIVE" }
                    });
                } else {
                    await tx.reservation.update({
                        where: { id: reservation.id },
                        data: { state: "POTENTIALLY_INVALID" }
                    });
                }
            }
 
 
            return prismaSpace;
        });
        // You should map to Space entity here if needed
        return spaceUpdated;
    }
}
 
export default SQLSpaceRepository;
